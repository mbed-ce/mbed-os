19#define NET_STACK_MEMORY_MANAGER_H

21/**
22 * Network Stack interface memory manager
23 *
24 * \par
25 * This interface provides abstraction for memory modules used in different IP stacks (often to accommodate zero
26 * copy). NetStack interface is required to accept output packets and provide received data using this stack-
27 * independent API. This header should be implemented for each IP stack, so that we keep EMAC module independent.
28 *
29 * \par
30 * NetStack memory interface uses memory buffer chains to store data. Data passed in either direction
31 * may either be contiguous (a single-buffer chain), or may consist of multiple buffers.
32 * Chaining of the buffers is made using singly-linked list. The NetStack data-passing APIs do not specify
33 * alignment or structure of the chain in either direction.
34 *
35 * \par
36 * Memory buffers can be allocated either from heap or from memory pools. Heap buffers are always contiguous.
37 * Memory pool buffers may be either contiguous or chained depending on allocation size. By LwIP convention,
38 * the pool should only be used for Rx packets -- the EMAC may opt to keep buffers pre-allocated from the pool
39 * for receiving packets into. This is done because LwIP will do special stuff when the pool runs out of space,
40 * e.g. flushing TCP out-of-sequence segment buffers to free up memory.
41 *
42 * On NetStack interface buffer chain ownership is transferred. EMAC must free buffer chain that it is given for
43 * link output and the stack must free the buffer chain that it is given for link input.
44 *
45 */
46

50typedef void net_stack_mem_buf_t; // Memory buffer
51
52class NetStackMemoryManager {
53protected:
54 /// Callback which shall be called (if set) by the implementation after one or more buffer spaces
55 /// become free in the pool. This is used by zero-copy Ethernet MACs as a hint that
56 /// now is a good time to allocate fresh buffers off the pool into Ethernet descriptors.
57 /// It *is* legal to call this function if you aren't totally sure new memory is available --
58 /// the mac will try to allocate more buffers, and if it can't, oh well.
59 /// However, it is not legal for memory to become available without a call to this function.
60 /// Such a situation might lead to a lockup of the MAC due to not having memory allocated for Rx.
61 // TODO this eventually needs to get converted to a list once we support boards with more than 1 EMAC
62 mbed::Callback<void()> onPoolSpaceAvailCallback;
63
64public:
65
66 /**
67 * Allocates memory buffer from the heap
68 *
69 * Memory buffer allocated from heap is always contiguous and can be arbitrary size.
70 *
71 * @param size Size of the memory to allocate in bytes
72 * @param align Memory alignment requirement in bytes
73 * @return Allocated memory buffer, or NULL in case of error
74 */
75 virtual net_stack_mem_buf_t *alloc_heap(uint32_t size, uint32_t align) = 0;
76
77 /**
78 * Allocates memory buffer chain from a pool
79 *
80 * Memory allocated from pool is contiguous if size is equal or less than
81 * (aligned) allocation unit, otherwise may be chained. Will typically come from
82 * fixed-size packet pool memory.
83 *
84 * @param size Total size of the memory to allocate in bytes
85 * @param align Memory alignment requirement for each buffer in bytes
86 * @return Allocated memory buffer chain, or NULL in case of error
87 */
88 virtual net_stack_mem_buf_t *alloc_pool(uint32_t size, uint32_t align) = 0;
89
90 /**
91 * Get memory buffer pool allocation unit
92 *
93 * Returns the maximum size of contiguous memory that can be allocated from a pool.
94 *
95 * @param align Memory alignment requirement in bytes
96 * @return Contiguous memory size
97 */
98 virtual uint32_t get_pool_alloc_unit(uint32_t align) const = 0;
99
100 /**
101 * Get memory buffer pool size.
102 *
103 * @return The number of pool buffers that can be allocated at any one time
104 */
105 uint32_t get_pool_size()
106 {
107#if UNITTEST
108 return 0;
109#else
110 return MBED_CONF_NSAPI_EMAC_RX_POOL_NUM_BUFS;
111#endif
112 }
113
114 /**
115 * Free memory buffer chain
116 *
117 * Frees all buffers from the chained list.
118 *
119 * @param buf Memory buffer chain to be freed.
120 */
121 virtual void free(net_stack_mem_buf_t *buf) = 0;
122
123 /**
124 * Return total length of a memory buffer chain
125 *
126 * Returns a total length of this buffer and any following buffers in the chain.
127 *
128 * @param buf Memory buffer chain
129 * @return Total length in bytes
130 */
131 virtual uint32_t get_total_len(const net_stack_mem_buf_t *buf) const = 0;
132
133 /**
134 * Copy a memory buffer chain
135 *
136 * Copies data from one buffer chain to another. Copy operation does not adjust the lengths
137 * of the copied-to memory buffer chain, so chain total lengths must be the same.
138 *
139 * @param to_buf Memory buffer chain to copy to
140 * @param from_buf Memory buffer chain to copy from
141 */
142 virtual void copy(net_stack_mem_buf_t *to_buf, const net_stack_mem_buf_t *from_buf) = 0;
143
144 /**
145 * Copy to a memory buffer chain
146 *
147 * Copies data to a buffer chain. Copy operation does not adjust the lengths
148 * of the copied-to memory buffer chain, so chain total length must match the
149 * copied length.
150 *
151 * @param to_buf Memory buffer chain to copy to
152 * @param ptr Pointer to data
153 * @param len Data length
154 */
155 virtual void copy_to_buf(net_stack_mem_buf_t *to_buf, const void *ptr, uint32_t len);
156
157 /**
158 * Copy from a memory buffer chain
159 *
160 * Copies data from a memory buffer chain.
161 *
162 * @param len Data length
163 * @param ptr Pointer to data
164 * @param from_buf Memory buffer chain to copy from
165 * @return Length of the data that was copied
166 */
167 virtual uint32_t copy_from_buf(void *ptr, uint32_t len, const net_stack_mem_buf_t *from_buf) const;
168
169 /**
170 * Concatenate two memory buffer chains
171 *
172 * Concatenates buffer chain to end of the other buffer chain. Concatenated-to buffer total length
173 * is adjusted accordingly. cat_buf must point to the start of a the chain. After concatenation
174 * to_buf's chain now owns those buffers, and they will be freed when the to_buf chain is freed.
175 *
176 * @param to_buf Memory buffer chain to concatenate to
177 * @param cat_buf Memory buffer chain to concatenate
178 */
179 virtual void cat(net_stack_mem_buf_t *to_buf, net_stack_mem_buf_t *cat_buf) = 0;
180
181 /**
182 * Returns the next buffer
183 *
184 * Returns the next buffer from the memory buffer chain.
185 *
186 * @param buf Memory buffer
187 * @return The next memory buffer, or NULL if last
188 */
189 virtual net_stack_mem_buf_t *get_next(const net_stack_mem_buf_t *buf) const = 0;
190
191 /**
192 * @brief Count the number of buffers in a buffer chain
193 *
194 * @param buf Memory buffer
195 * @return The number of buffers in the chain
196 */
197 size_t count_buffers(const net_stack_mem_buf_t *buf)
198 {
199 size_t count = 0;
200 while (buf != nullptr) {
201 count += 1;
202 buf = get_next(buf);
203 }
204 return count;
205 }
206
207 /**
208 * Return pointer to the payload of the buffer
209 *
210 * @param buf Memory buffer
211 * @return Pointer to the payload
212 */
213 virtual void *get_ptr(const net_stack_mem_buf_t *buf) const = 0;
214
215 /**
216 * Return payload size of this individual buffer (NOT including any chained buffers)
217 *
218 * @param buf Memory buffer
219 * @return Size in bytes
220 */
221 virtual uint32_t get_len(const net_stack_mem_buf_t *buf) const = 0;
222
223 /**
224 * Sets the payload size of the buffer
225 *
226 * The allocated payload size will not change. It is not permitted
227 * to change the length of a buffer that is not the first (or only) in a chain.
228 *
229 * *Note as of Dec 2024: Different implementations (Nanostack vs LwIP) disagree about
230 * how to implement this operation. Specifically, if called on the head of a buffer
231 * chain, the LwIP implementation allows changing the length of the chain as a whole.
232 * However, the Nanostack implementation does not and only can change the length of the head buffer.
233 * For fear of breaking existing code, I do not want to change this behavior.
234 * So, if constructing a buffer chain, it is safest to set the buffer lengths first before
235 * building the chain.
236 *
237 * @param buf Memory buffer
238 * @param len Payload size, must be less or equal to the allocated size
239 */
240 virtual void set_len(net_stack_mem_buf_t *buf, uint32_t len) = 0;
241
242 enum class Lifetime {
243 POOL_ALLOCATED, ///< Allocated from the memory manager's pool
244 HEAP_ALLOCATED, ///< Allocated from the memory manager's heap
245 CONSTANT, ///< Buffer points to constant data (e.g. in ROM) that will live forever
246 VOLATILE ///< Buffer points to data from the application that will not live past the current network stack call.
247 };
248
249 /**
250 * Gets the lifetime of the buffer
251 *
252 * @param buf Memory buffer
253 */
254 virtual Lifetime get_lifetime(net_stack_mem_buf_t const *buf) const = 0;
255
256 /**
257 * @brief Set callback which will be called when pool space becomes available
258 *
259 * \warning The callback could be called from any thread, and should make no assumptions about
260 * being in the same thread as anything else.
261 *
262 * @param cb Callback to call
263 */
264 void set_on_pool_space_avail_cb(mbed::Callback<void()> cb)
265 {
266 onPoolSpaceAvailCallback = cb;
267 }
268
269protected:
270 ~NetStackMemoryManager() = default;
271};
